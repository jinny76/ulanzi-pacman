# 吃豆人插件设计方案

## 一、概述

### 1.1 创意描述
创建一个动态的吃豆人游戏插件，吃豆人会在所有配置的按键之间移动，形成一个连贯的动画效果。每个按键显示游戏场景的一部分，所有按键组合起来就是一个完整的游戏画面。

### 1.2 核心特性
- 🎮 吃豆人在按键网格中自动移动
- 🍒 显示豆子、墙壁等游戏元素
- 👻 可选：添加幽灵追逐效果
- 🎬 流畅的逐帧动画（张嘴/闭嘴）
- ⚙️ 可配置移动速度、路径模式
- 🕹️ 可选：按键交互控制方向

## 三、技术架构

### 3.1 核心概念

```
物理按键网格 (3x3 示例):
┌─────┬─────┬─────┐
│ 0,0 │ 0,1 │ 0,2 │  每个格子是144x144px
├─────┼─────┼─────┤
│ 1,0 │ 1,1 │ 1,2 │  吃豆人在格子间移动
├─────┼─────┼─────┤
│ 2,0 │ 2,1 │ 2,2 │  格子显示对应的游戏画面
└─────┴─────┴─────┘

虚拟游戏画面 (432x432px):
┌───────────────────────┐
│                       │  完整的游戏场景
│     ●→               │  在Canvas上绘制
│  ・ ・ ・ ・ ・        │  然后切分给各个按键
│                       │
│        👻             │
└───────────────────────┘
```

### 3.3 数据流架构

```
主服务 (app.js)
  ├─ GameEngine (游戏引擎)
  │    ├─ gameState: 游戏状态（吃豆人位置、豆子分布）
  │    ├─ update(): 更新游戏逻辑（移动、碰撞检测）
  │    └─ render(): 绘制完整游戏画面到大Canvas
  │
  ├─ RenderManager (渲染管理器)
  │    ├─ splitCanvas(): 切分大Canvas为多个小块
  │    └─ updateAllKeys(): 更新所有按键的图标
  │
  └─ ACTION_CACHES (按键实例管理)
       └─ { context -> PacmanCell }
```

## 四、核心模块设计

### 4.1 游戏引擎 (GameEngine.js)

```javascript
class GameEngine {
  constructor(gridConfig) {
    this.config = gridConfig;
    this.pacman = {
      row: 0,
      col: 0,
      direction: 'right',
      frame: 0,
      mouthOpen: true
    };

    // 豆子地图（true表示有豆子）
    this.dots = this.initializeDots();

    // 移动路径队列
    this.pathQueue = this.generateRandomPath();

    this.frameCount = 0;
  }

  /**
   * 初始化豆子分布
   */
  initializeDots() {
    const dots = [];
    for (let r = 0; r < this.config.rows; r++) {
      dots[r] = [];
      for (let c = 0; c < this.config.cols; c++) {
        dots[r][c] = true; // 每个格子都有豆子
      }
    }
    return dots;
  }

  /**
   * 生成随机移动路径
   */
  generateRandomPath() {
    const directions = ['up', 'down', 'left', 'right'];
    const path = [];

    for (let i = 0; i < 20; i++) {
      const dir = directions[Math.floor(Math.random() * 4)];
      path.push(dir);
    }

    return path;
  }

  /**
   * 更新游戏状态（每帧调用）
   */
  update() {
    this.frameCount++;

    // 动画帧更新（每5帧切换一次张嘴/闭嘴）
    if (this.frameCount % 5 === 0) {
      this.pacman.mouthOpen = !this.pacman.mouthOpen;
      this.pacman.frame = (this.pacman.frame + 1) % 4;
    }

    // 位置更新（每30帧移动一格）
    if (this.frameCount % 30 === 0) {
      this.movePacman();
    }
  }

  /**
   * 移动吃豆人
   */
  movePacman() {
    // 获取下一个方向
    if (this.pathQueue.length === 0) {
      this.pathQueue = this.generateRandomPath();
    }

    const direction = this.pathQueue.shift();
    this.pacman.direction = direction;

    // 计算新位置
    let newRow = this.pacman.row;
    let newCol = this.pacman.col;

    switch (direction) {
      case 'up':
        newRow = Math.max(0, newRow - 1);
        break;
      case 'down':
        newRow = Math.min(this.config.rows - 1, newRow + 1);
        break;
      case 'left':
        newCol = Math.max(0, newCol - 1);
        break;
      case 'right':
        newCol = Math.min(this.config.cols - 1, newCol + 1);
        break;
    }

    // 吃掉豆子
    if (this.dots[newRow][newCol]) {
      this.dots[newRow][newCol] = false;
    }

    // 更新位置
    this.pacman.row = newRow;
    this.pacman.col = newCol;

    // 如果所有豆子都吃完了，重新生成
    if (this.allDotsEaten()) {
      this.dots = this.initializeDots();
    }
  }

  /**
   * 检查是否所有豆子都被吃掉
   */
  allDotsEaten() {
    for (let r = 0; r < this.config.rows; r++) {
      for (let c = 0; c < this.config.cols; c++) {
        if (this.dots[r][c]) return false;
      }
    }
    return true;
  }

  /**
   * 获取游戏状态（供渲染使用）
   */
  getState() {
    return {
      pacman: { ...this.pacman },
      dots: this.dots
    };
  }
}
```

### 4.2 渲染管理器 (RenderManager.js)

```javascript
class RenderManager {
  constructor(gridConfig) {
    this.config = gridConfig;

    // 创建完整游戏画面的Canvas
    this.fullCanvas = document.createElement('canvas');
    this.fullCanvas.width = gridConfig.cols * gridConfig.cellSize;
    this.fullCanvas.height = gridConfig.rows * gridConfig.cellSize;
    this.fullCtx = this.fullCanvas.getContext('2d');

    // 为每个格子创建小Canvas
    this.cellCanvases = {};
  }

  /**
   * 渲染完整游戏画面
   */
  renderFullScene(gameState) {
    const ctx = this.fullCtx;
    const { pacman, dots } = gameState;

    // 清空画布
    ctx.fillStyle = '#000000';
    ctx.fillRect(0, 0, this.fullCanvas.width, this.fullCanvas.height);

    // 绘制格子网格线（调试用）
    this.drawGrid(ctx);

    // 绘制豆子
    this.drawDots(ctx, dots);

    // 绘制吃豆人
    this.drawPacman(ctx, pacman);
  }

  /**
   * 绘制网格线
   */
  drawGrid(ctx) {
    ctx.strokeStyle = '#1a1a1a';
    ctx.lineWidth = 2;

    // 垂直线
    for (let col = 0; col <= this.config.cols; col++) {
      const x = col * this.config.cellSize;
      ctx.beginPath();
      ctx.moveTo(x, 0);
      ctx.lineTo(x, this.fullCanvas.height);
      ctx.stroke();
    }

    // 水平线
    for (let row = 0; row <= this.config.rows; row++) {
      const y = row * this.config.cellSize;
      ctx.beginPath();
      ctx.moveTo(0, y);
      ctx.lineTo(this.fullCanvas.width, y);
      ctx.stroke();
    }
  }

  /**
   * 绘制豆子
   */
  drawDots(ctx, dots) {
    ctx.fillStyle = '#FFB8AE';

    for (let row = 0; row < this.config.rows; row++) {
      for (let col = 0; col < this.config.cols; col++) {
        if (dots[row][col]) {
          const centerX = col * this.config.cellSize + this.config.cellSize / 2;
          const centerY = row * this.config.cellSize + this.config.cellSize / 2;

          ctx.beginPath();
          ctx.arc(centerX, centerY, 6, 0, Math.PI * 2);
          ctx.fill();
        }
      }
    }
  }

  /**
   * 绘制吃豆人
   */
  drawPacman(ctx, pacman) {
    const centerX = pacman.col * this.config.cellSize + this.config.cellSize / 2;
    const centerY = pacman.row * this.config.cellSize + this.config.cellSize / 2;
    const radius = 50;

    // 吃豆人颜色
    ctx.fillStyle = '#FFFF00';

    // 根据方向旋转
    const rotationMap = {
      'right': 0,
      'down': Math.PI / 2,
      'left': Math.PI,
      'up': -Math.PI / 2
    };
    const rotation = rotationMap[pacman.direction];

    // 嘴巴张开角度
    const mouthAngle = pacman.mouthOpen ? Math.PI / 6 : 0.01;

    ctx.save();
    ctx.translate(centerX, centerY);
    ctx.rotate(rotation);

    // 绘制吃豆人（缺口扇形）
    ctx.beginPath();
    ctx.arc(0, 0, radius, mouthAngle, Math.PI * 2 - mouthAngle);
    ctx.lineTo(0, 0);
    ctx.closePath();
    ctx.fill();

    // 眼睛
    ctx.fillStyle = '#000000';
    ctx.beginPath();
    ctx.arc(-10, -20, 5, 0, Math.PI * 2);
    ctx.fill();

    ctx.restore();
  }

  /**
   * 切分完整画面为多个格子
   */
  splitToBlocks(keyPositionMap) {
    const blocks = {};

    Object.entries(keyPositionMap).forEach(([key, position]) => {
      const { row, col } = position;

      // 创建小Canvas
      const cellCanvas = document.createElement('canvas');
      cellCanvas.width = this.config.cellSize;
      cellCanvas.height = this.config.cellSize;
      const cellCtx = cellCanvas.getContext('2d');

      // 从完整画面中裁剪对应区域
      cellCtx.drawImage(
        this.fullCanvas,
        col * this.config.cellSize,
        row * this.config.cellSize,
        this.config.cellSize,
        this.config.cellSize,
        0, 0,
        this.config.cellSize,
        this.config.cellSize
      );

      blocks[key] = cellCanvas.toDataURL('image/png');
    });

    return blocks;
  }
}
```

### 4.3 主服务逻辑 (app.js)

```javascript
/**
 * 吃豆人插件主服务
 */

// 全局状态
const ACTION_CACHES = {};
let gameEngine = null;
let renderManager = null;
let updateTimer = null;

// 按键位置映射（从全局设置中加载）
let keyPositionMap = {};

// 网格配置
const GRID_CONFIG = {
  rows: 3,
  cols: 3,
  cellSize: 144
};

// 连接
$UD.connect('com.ulanzi.pacman');

$UD.onConnected(() => {
  Utils.log('[PacMan] Connected');

  // 获取全局设置
  $UD.getGlobalSettings();

  // 初始化游戏引擎
  gameEngine = new GameEngine(GRID_CONFIG);
  renderManager = new RenderManager(GRID_CONFIG);

  // 启动游戏循环
  startGameLoop();
});

/**
 * 启动游戏循环
 */
function startGameLoop() {
  if (updateTimer) clearInterval(updateTimer);

  // 60 FPS
  updateTimer = setInterval(() => {
    // 更新游戏逻辑
    gameEngine.update();

    // 渲染游戏画面
    const gameState = gameEngine.getState();
    renderManager.renderFullScene(gameState);

    // 切分并更新所有按键
    const blocks = renderManager.splitToBlocks(keyPositionMap);
    updateAllKeys(blocks);
  }, 1000 / 60);
}

/**
 * 更新所有按键的图标
 */
function updateAllKeys(blocks) {
  Object.entries(ACTION_CACHES).forEach(([context, instance]) => {
    if (!instance.allowSend) return;

    const { key } = $UD.decodeContext(context);
    const iconData = blocks[key];

    if (iconData) {
      $UD.setBaseDataIcon(context, iconData);
    }
  });
}

/**
 * Action 添加
 */
$UD.onAdd(jsn => {
  const context = jsn.context;

  if (!ACTION_CACHES[context]) {
    ACTION_CACHES[context] = {
      context,
      allowSend: true,
      position: null
    };
  }

  // 应用设置
  if (jsn.param) {
    onSetSettings(jsn);
  }
});

/**
 * 设置活跃状态
 */
$UD.onSetActive(jsn => {
  const instance = ACTION_CACHES[jsn.context];
  if (instance) {
    instance.allowSend = jsn.active;
  }
});

/**
 * 按键按下（可用于暂停/继续游戏）
 */
$UD.onRun(jsn => {
  const instance = ACTION_CACHES[jsn.context];
  if (!instance) return;

  // 切换游戏暂停状态
  if (updateTimer) {
    clearInterval(updateTimer);
    updateTimer = null;
    $UD.toast('游戏暂停');
  } else {
    startGameLoop();
    $UD.toast('游戏继续');
  }
});

/**
 * 清除 Action
 */
$UD.onClear(jsn => {
  if (jsn.param) {
    jsn.param.forEach(item => {
      delete ACTION_CACHES[item.context];
      delete keyPositionMap[item.key];
    });
  }
});

/**
 * 接收全局设置
 */
$UD.onDidReceiveGlobalSettings(jsn => {
  const settings = jsn.param || {};

  if (settings.grid_rows) {
    GRID_CONFIG.rows = parseInt(settings.grid_rows);
  }
  if (settings.grid_cols) {
    GRID_CONFIG.cols = parseInt(settings.grid_cols);
  }

  Utils.log('[PacMan] Grid config:', GRID_CONFIG);
});

/**
 * 更新设置
 */
function onSetSettings(jsn) {
  const settings = jsn.param || {};
  const context = jsn.context;
  const instance = ACTION_CACHES[context];

  if (!instance) return;

  // 保存按键位置信息
  if (settings.position_row !== undefined && settings.position_col !== undefined) {
    const { key } = $UD.decodeContext(context);
    keyPositionMap[key] = {
      row: parseInt(settings.position_row),
      col: parseInt(settings.position_col)
    };

    instance.position = keyPositionMap[key];
    Utils.log(`[PacMan] Key ${key} positioned at (${instance.position.row}, ${instance.position.col})`);
  }
}

/**
 * 参数更新
 */
$UD.onParamFromApp(onSetSettings);
$UD.onParamFromPlugin(onSetSettings);
```

### 5.2 配置逻辑 (inspector.js)

```javascript
let currentSettings = {};

$UD.connect();

$UD.onConnected(() => {
  const wrapper = document.querySelector('.uspi-wrapper');
  wrapper.classList.remove('hidden');

  // 绑定表单变化事件
  const inputs = document.querySelectorAll('input, select');
  inputs.forEach(input => {
    input.addEventListener('change', Utils.debounce(() => {
      sendSettings();
    }, 300));
  });
});

$UD.onAdd(jsn => {
  if (jsn.param) {
    loadSettings(jsn.param);
  }
});

$UD.onParamFromApp(jsn => {
  if (jsn.param) {
    loadSettings(jsn.param);
  }
});

function loadSettings(settings) {
  currentSettings = settings;

  if (settings.position_row !== undefined) {
    document.querySelector('[name="position_row"]').value = settings.position_row;
  }
  if (settings.position_col !== undefined) {
    document.querySelector('[name="position_col"]').value = settings.position_col;
  }
  if (settings.speed) {
    document.querySelector('[name="speed"]').value = settings.speed;
  }
  if (settings.path_mode) {
    document.querySelector('[name="path_mode"]').value = settings.path_mode;
  }
}

function sendSettings() {
  const settings = {
    position_row: parseInt(document.querySelector('[name="position_row"]').value),
    position_col: parseInt(document.querySelector('[name="position_col"]').value),
    speed: document.querySelector('[name="speed"]').value,
    path_mode: document.querySelector('[name="path_mode"]').value
  };

  $UD.sendParamFromPlugin(settings);
}
```

## 六、使用流程

### 6.1 用户操作步骤

1. **安装插件**：将插件复制到 UlanziStudio 插件目录

2. **布局按键**：
   - 将 "Pac-Man Runner" Action 拖到多个按键上
   - 建议使用 3x3 或 2x5 的布局

3. **配置位置**：
   - 点击每个按键，打开配置界面
   - 设置该按键在网格中的位置（行、列）
   ```
   示例：3x3网格
   按键A → 行:0, 列:0 (左上角)
   按键B → 行:0, 列:1 (上中)
   按键C → 行:0, 列:2 (右上角)
   ...
   ```

4. **享受动画**：配置完成后，吃豆人会自动在按键之间移动

## 八、技术要点总结

### 8.1 关键技术

1. **全局Canvas渲染**
   - 创建大Canvas绘制完整场景
   - 按网格切分为小块
   - 分发给各个按键

2. **状态同步**
   - 主服务维护游戏状态
   - 定时更新所有按键
   - 通过context识别每个按键

3. **坐标映射**
   - 用户配置key到坐标的映射
   - 游戏引擎基于逻辑坐标
   - 渲染管理器处理像素坐标转换

4. **动画流畅性**
   - 60 FPS 更新频率
   - 逐帧动画（张嘴/闭嘴）
   - 平滑移动过渡

### 8.2 性能优化

- Canvas复用，减少创建销毁
- 只更新活跃（allowSend=true）的按键
- 避免在循环中频繁调用SDK API
